// Import the fs module from Node.js to work with the file system const fs = require('node:fs'); // Define an array containing numbers representing task IDs const numbers = [...Array(10).keys(), 100, 101, 102, 103]; // Initialize an empty string to store the HTML list items let list = '
    '; // Iterate over each task number for (const number of numbers) { // Create a list item with a hyperlink to the corresponding task HTML file list += `
  • Task ${number}
    • `; } // Close the unordered list tag list += '
'; // Generate the HTML content for the index page const indexHTML = ` Holberton School
All tasks
${list} `; // Write the generated HTML content to the index.html file fs.writeFileSync('index.html', indexHTML); // Log a message to indicate that the index.html file has been successfully generated console.log('index.html file generated successfully.');
